import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load inbuilt Titanic dataset
df = sns.load_dataset("titanic")
print("Titanic Dataset Loaded Successfully")

print(df.head())

print("\nData Types:")
print(df.dtypes)

print("\nMissing Values:")
print(df.isnull().sum())

# Fill missing age values
df['age'].fillna(df['age'].median())

# Boxplot: Age distribution by gender
sns.boxplot(x='sex', y='age', data=df)
plt.title("Age Distribution by Gender")
plt.show()

# Boxplot: Age distribution by gender and survival
sns.boxplot(x='sex', y='age', hue='survived', data=df)
plt.title("Age Distribution by Gender and Survival")
plt.show()

# Observations
print("\nObservations:")
print("1. Females had higher survival rate.")
print("2. Younger passengers were more likely to survive.")
print("3. Boxplot helps identify outliers and age distribution.")
